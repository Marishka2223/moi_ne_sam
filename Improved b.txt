        <script>
            function checkPassword(form) {
                const password1 = form.password.value;
                const password2 = form.password_d.value;

                if (password1 === '') {
                    alert("Введите пароль");
                    return false;
                } else if (password2 === '') {
                    alert("Подтвердите пароль");
                    return false;
                } else if (password1 !== password2) {
                    alert("Пароли не совпадают.");
                    return false;
                }
                return true;
            }
        </script>
