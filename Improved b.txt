
<?php

// Имитация класса модели Bb
class Bb
{
    public static function findOrFail($id)
    {
        // Пример данных для подписки
        $subscriptions = [
            1 => ['id' => 1, 'title' => 'Duck Basic', 'price' => 3, 'status' => 'Активно'],
            2 => ['id' => 2, 'title' => 'Duck+', 'price' => 6, 'status' => 'Активно'],
            3 => ['id' => 3, 'title' => 'Duck PREMIUM', 'price' => 12, 'status' => 'Активно'],
        ];

        if (isset($subscriptions[$id])) {
            return (object) $subscriptions[$id];
        }

        throw new Exception("Подписка не найдена");
    }
}

// Имитация класса контроллера
class UpgradePlanController
{
    // Показать доступные варианты улучшения
    public function showUpgradeOptions($id)
    {
        // Получаем текущую подписку
        $subscription = Bb::findOrFail($id);

        // Определяем доступные варианты улучшения
        $availablePlans = [];
        if ($subscription->price == 3) {
            $availablePlans = [
                ['title' => 'Duck+', 'price' => 6],
                ['title' => 'Duck PREMIUM', 'price' => 12],
            ];
        } elseif ($subscription->price == 6) {
            $availablePlans = [
                ['title' => 'Duck PREMIUM', 'price' => 12],
            ];
        }

        // Рендерим представление
        return $this->view('upgrade-plan', [
            'subscription' => $subscription,
            'availablePlans' => $availablePlans,
        ]);
    }

    // Обработка улучшения плана
    public function upgradePlan($id, $request)
    {
        // Валидация данных
        if (empty($request['title']) || !in_array($request['price'], [6, 12])) {
            throw new Exception("Неверные данные");
        }

        // Обновляем подписку
        $subscription = Bb::findOrFail($id);
        $subscription->title = $request['title'];
        $subscription->price = $request['price'];
        $subscription->status = 'Активно';

        // Редирект (в реальном проекте это было бы через header())
        echo "План успешно улучшен!";
        exit;
    }

    // Простая функция для рендеринга представления
    private function view($template, $data = [])
    {
        extract($data);
        ob_start();
        include $template . '.php';
        return ob_get_clean();
    }
}

// Имитация маршрутизации
$uri = $_SERVER['REQUEST_URI'];
$method = $_SERVER['REQUEST_METHOD'];

if ($method === 'GET' && preg_match('/^\/upgrade-plan\/(\d+)$/', $uri, $matches)) {
    $controller = new UpgradePlanController();
    echo $controller->showUpgradeOptions($matches[1]);
} elseif ($method === 'POST' && preg_match('/^\/upgrade-plan\/(\d+)$/', $uri, $matches)) {
    $controller = new UpgradePlanController();
    $controller->upgradePlan($matches[1], $_POST);
} else {
    echo "Страница не найдена";
    exit;
}
