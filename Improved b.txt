<?php

// Имитация модели (для примера)
class Bb
{
    public $id;
    public $title;
    public $price;
    public $status;

    public static function findOrFail($id)
    {
        // Имитация поиска подписки в базе данных
        $subscription = new self();
        $subscription->id = $id;
        $subscription->title = 'Текущий план';
        $subscription->price = 3; // Имитация текущей цены
        $subscription->status = 'Активно';
        return $subscription;
    }

    public function update($data)
    {
        // Имитация обновления подписки
        foreach ($data as $key => $value) {
            $this->$key = $value;
        }
    }
}

// Имитация контроллера
class UpgradePlanController
{
    // Показать доступные варианты улучшения
    public function showUpgradeOptions($id)
    {
        // Получаем текущую подписку
        $subscription = Bb::findOrFail($id);

        // Определяем доступные варианты улучшения
        $availablePlans = [];
        if ($subscription->price == 3) {
            $availablePlans = [
                ['title' => 'Duck+', 'price' => 6],
                ['title' => 'Duck PREMIUM', 'price' => 12],
            ];
        } elseif ($subscription->price == 6) {
            $availablePlans = [
                ['title' => 'Duck PREMIUM', 'price' => 12],
            ];
        }

        // Рендерим представление
        $this->renderView('upgrade-plan', [
            'subscription' => $subscription,
            'availablePlans' => $availablePlans,
        ]);
    }

    // Обработка улучшения плана
    public function upgradePlan($id)
    {
        // Имитация данных из POST-запроса
        $title = $_POST['title'];
        $price = $_POST['price'];

        // Валидация данных
        if (empty($title) || !in_array($price, [6, 12])) {
            die('Ошибка валидации данных.');
        }

        // Обновляем подписку
        $subscription = Bb::findOrFail($id);
        $subscription->update([
            'title' => $title,
            'price' => $price,
            'status' => 'Активно',
        ]);

        // Перенаправляем пользователя (имитация)
        header('Location: /user/subscriptions');
        exit;
    }

    // Функция для рендеринга представления
    private function renderView($view, $data = [])
    {
        extract($data);
        include __DIR__ . "/$view.php";
    }
}

// Имитация маршрутизации
if ($_SERVER['REQUEST_METHOD'] === 'GET' && isset($_GET['id'])) {
    $controller = new UpgradePlanController();
    $controller->showUpgradeOptions($_GET['id']);
} elseif ($_SERVER['REQUEST_METHOD'] === 'POST' && isset($_POST['title']) && isset($_POST['price'])) {
    $controller = new UpgradePlanController();
    $controller->upgradePlan($_POST['id']);
} else {
    die('Неверный запрос.');
}
