<?php
session_start(); // Начинаем сессию

// Проверяем, был ли отправлен товар
if ($_SERVER['REQUEST_METHOD'] === 'POST' && isset($_POST['product'])) {
    $selectedProduct = $_POST['product'];
    
    // Сохраняем выбранный товар в сессии
    if (!isset($_SESSION['cart'])) {
        $_SESSION['cart'] = []; // Создаем массив корзины, если его нет
    }
    
    $_SESSION['cart'][] = $selectedProduct; // Добавляем товар в корзину
}

?>

<!DOCTYPE html>
<html lang="ru">
<head>
    <meta charset="UTF-8">
    <title>Выбор товара</title>
</head>
<body>
    <h1>Выберите товар</h1>
    
    <form action="zakaz2.php" method="post">
        <select name="product">
            <option value="Товар 1">Товар 1</option>
            <option value="Товар 2">Товар 2</option>
            <option value="Товар 3">Товар 3</option>
        </select>
        <button type="submit">Добавить в корзину</button>
    </form>
    
    <p><a href="cart.php">Перейти в корзину</a></p>
</body>

<?php
session_start(); // Начинаем сессию

// Проверяем, есть ли товары в корзине
if (isset($_SESSION['cart']) && !empty($_SESSION['cart'])) {
    echo "<h1>Ваша корзина</h1>";
    echo "<ul>";
    foreach ($_SESSION['cart'] as $product) {
        echo "<li>$product</li>";
    }
    echo "</ul>";
} else {
    echo "<h1>Ваша корзина пуста</h1>";
}
?>

<p><a href="zakaz.php">Вернуться к выбору товаров</a></p>
